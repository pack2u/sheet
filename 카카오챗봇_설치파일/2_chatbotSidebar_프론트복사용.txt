      <button id="send-btn" onclick="sendMessage()">전송</button>

    <script>
      function handleEnter(e) {
        if (e.key === 'Enter') {
          sendMessage();
        }
      }

      function scrollToBottom() {
        var chatArea = document.getElementById('chat-area');
        chatArea.scrollTop = chatArea.scrollHeight;
      }

      function sendMessage() {
        var inputObj = document.getElementById('user-input');
        var btnObj = document.getElementById('send-btn');
        var text = inputObj.value.trim();
        
        if (text === '') return;
        
        inputObj.value = '';
        inputObj.disabled = true;
        btnObj.disabled = true;

        // 1. 사용자 질문을 화면에 표시
        var chatArea = document.getElementById('chat-area');
        chatArea.innerHTML += '<div class="message message-user"><div class="bubble bubble-user">' + text + '</div></div>';
        
        // 2. 로딩 애니메이션 표시
        var loadingId = 'loading-' + new Date().getTime();
        chatArea.innerHTML += '<div class="message message-bot" id="' + loadingId + '"><div class="bubble bubble-bot">매뉴얼 검토 중...💬</div></div>';
        scrollToBottom();

        // 3. 구글 스크립트(서버) 통신
        google.script.run
          .withSuccessHandler(function(response) {
            var loadingElem = document.getElementById(loadingId);
            if (loadingElem) loadingElem.remove();
            
            chatArea.innerHTML += '<div class="message message-bot"><div class="bubble bubble-bot">' + response + '</div></div>';
            scrollToBottom();
            
            inputObj.disabled = false;
            btnObj.disabled = false;
            inputObj.focus();
          })
          .withFailureHandler(function(error) {
            var loadingElem = document.getElementById(loadingId);
            if (loadingElem) loadingElem.remove();
            
            chatArea.innerHTML += '<div class="message message-bot"><div class="bubble bubble-bot" style="color:red;">통신 에러: ' + error.message + '</div></div>';
            scrollToBottom();
            
            inputObj.disabled = false;
            btnObj.disabled = false;
          })
          .callGeminiAPI(text);
      }
    </script>
